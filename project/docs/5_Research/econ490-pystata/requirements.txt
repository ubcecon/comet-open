# Python packages for the ECON 490 PyStata notebooks.
# Install with: pip install -r requirements.txt

# Note: pystata doesn't come from pip. It ships with Stata itself, inside your
# Stata folder (for example D:/Stata/utilities/pystata). stata_setup is just the
# helper that adds that folder to Python's path so the notebooks can find it.
# UBC students can get Stata for free: https://isit.arts.ubc.ca/news/stata-18-available-now/

# A few things you'll need that pip can't install for you:
#   - Stata SE 18 or 19 installed locally, with a valid license
#   - Quarto 1.5 or newer (https://quarto.org) to render the .qmd files
#   - A Jupyter Python kernel (ipykernel below registers one)

stata_setup>=0.1.3      # finds the pystata that ships with Stata
numpy>=1.23             # needed by pystata

ipython>=8.0            # gives us the %%stata cell magic
jupyter                 # quarto runs cells through the jupyter engine
ipykernel               # the python3 kernel quarto launches
nbformat                # notebook handling for quarto/jupyter
nbclient                # notebook execution for quarto/jupyter
